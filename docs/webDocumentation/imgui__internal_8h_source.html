487inline bool ImBitArrayTestBit(const ImU32* arr, int n) { ImU32 mask = (ImU32)1 << (n & 31); return (arr[n >> 5] & mask) != 0; }
488inline void ImBitArrayClearBit(ImU32* arr, int n) { ImU32 mask = (ImU32)1 << (n & 31); arr[n >> 5] &= ~mask; }
489inline void ImBitArraySetBit(ImU32* arr, int n) { ImU32 mask = (ImU32)1 << (n & 31); arr[n >> 5] |= mask; }
490inline void ImBitArraySetBitRange(ImU32* arr, int n, int n2) // Works on range [n..n2)
491{
492 n2--;
493 while (n <= n2)
494 {
495 int a_mod = (n & 31);
496 int b_mod = (n2 > (n | 31) ? 31 : (n2 & 31)) + 1;
497 ImU32 mask = (ImU32)(((ImU64)1 << b_mod) - 1) & ~(ImU32)(((ImU64)1 << a_mod) - 1);
498 arr[n >> 5] |= mask;
499 n = (n + 32) & ~31;
500 }
501}

560// Facilitate storing multiple chunks into a single large block (the "arena")

564{

576 inline void* GetSpanPtrEnd(int n) { IM_ASSERT(n >= 0 && n < CHUNKS && CurrIdx == CHUNKS); return (void*)(BasePtr + Offsets[n] + Sizes[n]); }
